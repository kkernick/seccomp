2//! A wrapper for the SECCOMP Notify interface.
3//!
4//! ## Implementation
5//! This implementation does not make any assumptions about how you get the
6//! SECCOMP FD. However, there are some considerations you should take into account:
7//! * SECCOMP applies across threads. If you place the monitor in a separate thread,
8//!   making sending the FD easy, the monitor will be confined by the policy it's monitoring.
9//!   This can cause deadlock where the monitor uses a syscall, which the kernel sends an
10//!   event for, to which the monitor cannot handle because its currently waiting for itself
11//!   to request it.
12//! * FDs can be passed across a socket, but you cannot get the SECCOMP FD until you have
13//!   loaded the filter. This means you need to ensure that the syscalls used to send the
14//!   FD (`connect`, `sendmsg`, etc) are not sent to the notifier, who does not have
15//!   the SECCOMP FD yet. `fd_socket` provides functions to send and receive a FD between
16//!   processes. See `antimony-monitor`, and Antimony as a whole, to see how you can
17//!   notify safely (Hint: Notify all Syscalls except those needed to send FD, which are
18//!   instead logged on Audit, with a separate thread for reading the log).
19

28/// Errors regarding to Notify.
29#[derive(Debug)]
30pub enum Error {
31    /// If the pair cannot be allocated.
32    Allocation(Errno),
33
34    /// If there was an error receiving a request from the kernel.
35    Receive(Errno),
36
37    /// If there was an error sending a response to a request.
38    Respond(Errno),
39}
40impl error::Error for Error {
41    fn source(&self) -> Option<&(dyn error::Error + 'static)> {
42        match self {
43            Self::Allocation(errno) | Self::Receive(errno) => Some(errno),
44            Self::Respond(_) => None,
45        }
46    }
47}
48impl fmt::Display for Error {
49    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
50        match self {
51            Self::Allocation(errno) => write!(f, "Failed to allocate notification pair: {errno}"),
52            Self::Receive(errno) => write!(f, "Failed to receive event: {errno}"),
53            Self::Respond(errno) => write!(f, "Failed to respond to event: {errno}"),
54        }
55    }
56}

58/// A trait for transmitting a SECCOMP Notify FD to a Monitor.
59///
60/// Executors, such as `spawn`, should perform the following actions
61/// from the Filter.
62///
63/// 1. Call `Notifier::exempt()`
64/// 2. Call `Notifier::prepare()`
65/// 3. Call `seccomp_load()`
66/// 4. Call `Notifier::handle()`
67///
68/// Then, call `execve()`.
69/// See Antimony for a socket implementation.
70pub trait Notifier: Send + 'static {
71    /// Return the list of syscalls that are used by the Notifier itself
72    /// in order to transmit the SECCOMP FD. These syscalls will be used
73    /// between `seccomp_load()` and `execve()`. For example, if sending
74    /// the FD across a socket, you should pass `sendmsg`.
75    ///
76    /// The action should NOT be Notify, as that will cause a deadlock.
77    /// Instead, either Allow, or Log.
78    fn exempt(&self) -> Vec<(Action, Syscall)> {
79        Vec::new()
80    }
81
82    /// Prepare for `seccomp_load`. This function is the last thing run
83    /// before `seccomp_load`, and as such is the last time you will
84    /// not be confined by the Filter. This can be used, for example,
85    /// to wait for a socket, then connect to it.
86    ///
87    /// ## Errors
88    /// This function should fail if preparation could not be made.
89    fn prepare(&mut self) -> Result<(), String> {
90        Ok(())
91    }
92
93    /// Handle the SECCOMP FD. This function runs under the confined
94    /// SECCOMP Filter, and should transmit the `OwnedFD` to the
95    /// Notify Monitor. The more you do here, the more syscalls
96    /// you will need; consider moving as much as possible to
97    /// `prepare()`
98    fn handle(&mut self, fd: OwnedFd);
99}

101/// A Notification Pair.
102///
103/// ## Examples
104///
105/// ```rust,ignore
106/// let pair = seccomp::notify::Pair::new().unwrap();
107/// loop {
108///     match pair.recv(raw) {
109///         Ok(Some(_)) => pair.reply(raw, |req, resp| {
110///             resp.val = 0;
111///
112///             // Deny syscall 1.
113///             resp.error = if req.data.nr == 1 {
114///                 EPERM
115///             } else {
116///                 0
117///             };
118///
119///             // Allow everything else.
120///             resp.flags = 1;
121///         }).unwrap(),
122///         Ok(None) => continue,
123///         Err(_) => break
124///     }
125/// }
126/// ```
127pub struct Pair {
128    /// The structure filled by the kernel on new events.
129    req: *mut raw::seccomp_notif,
130
131    /// The constructed response to send back.
132    resp: *mut raw::seccomp_notif_resp,
133}
134impl Pair {
135    /// Construct a new Pair.
136    ///
137    /// ## Errors
138    /// `Error::Allocation`: if the notify pair could not be allocated.
139    pub fn new() -> Result<Self, Error> {
140        let (req, resp) = unsafe {
141            let mut req: *mut raw::seccomp_notif = null_mut();
142            let mut resp: *mut raw::seccomp_notif_resp = null_mut();
143            match raw::seccomp_notify_alloc(&raw mut req, &raw mut resp) {
144                0 => (req, resp),
145                e => return Err(Error::Allocation(Errno::from_raw(e))),
146            }
147        };
148        Ok(Self { req, resp })
149    }
150
151    /// Receive a new event.
152    /// This function fails if the kernel returns an error.
153    ///
154    /// ## Errors
155    /// `Error::Receive`: If `seccomp_notify_receive` returned an error.
156    pub fn recv(&self, fd: RawFd) -> Result<Option<()>, Error> {
157        // We need to wipe the structure each time.
158        unsafe {
159            ptr::write_bytes(self.req, 0, 1);
160        }
161        // Call seccomp_notify_receive
162        let ret = unsafe { raw::seccomp_notify_receive(fd, self.req) };
163        if ret < 0 {
164            match Errno::last() {
165                Errno::EINTR | Errno::EAGAIN | Errno::ENOENT => Ok(None),
166                err => Err(Error::Receive(err)),
167            }
168        } else {
169            Ok(Some(()))
170        }
171    }
172
173    /// Reply to the last event.
174    ///
175    /// ## Handle
176    /// Handle offloads the actual decision of the request to your application.
177    /// It takes a constant reference to the event from the kernel, and a mutable
178    /// reference to the response. Parse the former to populate the latter, and
179    /// the Pair will send the response over.
180    ///
181    /// The request will always be valid, and the ID will already be set.
182    ///
183    /// ## Errors
184    /// `Error::Respond` if the `seccomp_notify_respond` fails
185    pub fn reply<F>(&self, fd: RawFd, handle: F) -> Result<(), Error>
186    where
187        F: Fn(&raw::seccomp_notif, &mut raw::seccomp_notif_resp),
188    {
189        let (req, resp) = unsafe { (&*self.req, &mut *self.resp) };
190
191        // Ensure the request is still valid.
192        let valid = unsafe { raw::seccomp_notify_id_valid(fd, req.id) };
193        if valid != 0 {
194            return Ok(());
195        }
196
197        // Set the ID.
198        resp.id = req.id;
199
200        // Delegate the decision to the closure.
201        handle(req, resp);
202
203        // Send response
204        let respond_ret = unsafe { raw::seccomp_notify_respond(fd.as_raw_fd(), self.resp) };
205        if respond_ret < 0 {
206            respond_ret
207                .checked_mul(-1)
208                .map_or(
209                    Err(Error::Respond(Errno::EINVAL)),
210                    |errno| match Errno::from_raw(errno) {
211                        Errno::ECANCELED => Ok(()),
212                        e => Err(Error::Respond(e)),
213                    },
214                )
215        } else {
216            Ok(())
217        }
218    }
219}
220impl Drop for Pair {
221    fn drop(&mut self) {
222        unsafe { raw::seccomp_notify_free(self.req, self.resp) }
223    }
224}
225// The Notify API is Thread Safe, and we're moving the Pair anyways.
226unsafe impl Send for Pair {}
